    <script>
        function fakeNavClick(el) {
            document.querySelectorAll('nav a').forEach(a => a.classList.remove('active', 'bg-gray-100', 'dark:bg-gray-900'));
            el.classList.add('active', 'bg-gray-100', 'dark:bg-gray-900');
            if (el.textContent.includes('Home')) return;
            alert('📍 This would navigate to ' + el.textContent + ' page');
        }

        function composePost() {
            document.getElementById('compose-text').focus();
        }

        function postTweet() {
            const text = document.getElementById('compose-text').value.trim();
            if (!text) return alert("Write something to post!");
            
            const feed = document.getElementById('feed');
            const newPost = document.createElement('div');
            newPost.className = "post-hover px-6 py-4 border-b border-gray-200 dark:border-gray-800 flex gap-4 animate-fade-in";
            newPost.innerHTML = `
                <div class="w-10 h-10 bg-gradient-to-br from-blue-500 to-purple-500 rounded-2xl flex-shrink-0 flex items-center justify-center text-white font-bold">A</div>
                <div class="flex-1">
                    <div class="flex items-center gap-2">
                        <span class="font-bold">Afu</span>
                        <span class="text-gray-500">@afu_ug</span>
                        <span class="text-gray-500">· now</span>
                    </div>
                    <p class="text-xl leading-tight mt-1">${text}</p>
                    <div class="flex justify-between mt-4 text-gray-500 max-w-[425px]">
                        <div class="flex items-center gap-3 hover:text-[#1d9bf0] cursor-pointer"><span>💭</span><span>0</span></div>
                        <div class="flex items-center gap-3 hover:text-[#1d9bf0] cursor-pointer"><span>🔁</span><span>0</span></div>
                        <div class="flex items-center gap-3 hover:text-[#1d9bf0] cursor-pointer"><span>❤️</span><span>1</span></div>
                        <div class="flex items-center gap-3 hover:text-[#1d9bf0] cursor-pointer"><span>📊</span><span>42</span></div>
                    </div>
                </div>
            `;
            feed.insertBefore(newPost, feed.firstChild);
            document.getElementById('compose-text').value = '';
            alert('✅ Posted live on your X home feed!');
        }

        function switchTab(n) {
            const foryou = document.getElementById('tab-foryou');
            const following = document.getElementById('tab-following');
            foryou.dataset.active = n === 0 ? 'true' : 'false';
            following.dataset.active = n === 1 ? 'true' : 'false';
        }
