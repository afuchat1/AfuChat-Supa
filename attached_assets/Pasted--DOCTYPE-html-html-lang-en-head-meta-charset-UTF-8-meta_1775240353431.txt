    <script>
        function switchPage(page) {
            const loginPage = document.getElementById('page-login');
            const signupPage = document.getElementById('page-signup');
            const tabLogin = document.getElementById('tab-login');
            const tabSignup = document.getElementById('tab-signup');
            
            if (page === 0) {
                loginPage.classList.remove('hidden');
                signupPage.classList.add('hidden');
                tabLogin.dataset.active = 'true';
                tabSignup.dataset.active = 'false';
            } else {
                loginPage.classList.add('hidden');
                signupPage.classList.remove('hidden');
                tabLogin.dataset.active = 'false';
                tabSignup.dataset.active = 'true';
            }
        }

        function fakeLogin(provider) {
            alert(`✅ ${provider} OAuth flow would start here (real X behavior)`);
        }

        function nextStepLogin() {
            const input = document.getElementById('login-input').value.trim();
            if (!input) return alert("Enter your phone, email or username");
            alert("✅ Password screen would appear next (multi-step flow simulated)");
        }

        function nextStepSignup() {
            const name = document.getElementById('signup-name').value.trim();
            const input = document.getElementById('signup-input').value.trim();
            if (!name || !input) return alert("Please fill in Name and Phone/Email");
            alert("✅ Next step: Birthday + Password flow would appear (exact X sign-up behavior)");
        }

        // Auto dark mode
        if (window.matchMedia('(prefers-color-scheme: dark)').matches) {
            document.documentElement.classList.add('dark');
        }

        console.log('%c✅ Both X Desktop Auth Pages loaded!', 'color:#1d9bf0; font-size:15px; font-weight:bold');
    </script>
