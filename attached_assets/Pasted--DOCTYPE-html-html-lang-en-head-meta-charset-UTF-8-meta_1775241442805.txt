    <script>
        let currentChat = -1;

        const chats = [
            {
                name: "Afu (You)",
                avatar: "🇺🇬",
                status: "online",
                messages: [
                    { type: "received", text: "Hey, the X UI clone looked dope!", time: "14:20" },
                    { type: "sent", text: "Thanks! Now building WhatsApp Web too 🔥", time: "14:21" },
                    { type: "received", text: "Legend", time: "14:22" }
                ]
            },
            {
                name: "Elon Musk",
                avatar: "🚀",
                status: "last seen recently",
                messages: [
                    { type: "received", text: "Grok 4 is insane", time: "14:22" },
                    { type: "sent", text: "Agreed! Just used it to build this UI", time: "14:23" }
                ]
            },
            {
                name: "Mumo • Kenya",
                avatar: "🇰🇪",
                status: "online",
                messages: [
                    { type: "received", text: "When are you coming to Nairobi?", time: "Yesterday" },
                    { type: "sent", text: "Soon! Let's build something crazy together", time: "Yesterday" }
                ]
            }
        ];

        function openChat(index) {
            currentChat = index;
            const chat = chats[index];
            
            // Update header
            document.getElementById('header-name').textContent = chat.name;
            document.getElementById('header-status').textContent = chat.status;
            document.getElementById('header-avatar').innerHTML = `<div class="w-9 h-9 flex items-center justify-center text-3xl">${chat.avatar}</div>`;
            
            // Update info panel
            document.getElementById('info-name').textContent = chat.name;
            document.getElementById('info-avatar').innerHTML = `<div class="w-24 h-24 flex items-center justify-center text-6xl">${chat.avatar}</div>`;
            
            // Render messages
            const messagesContainer = document.getElementById('messages');
            messagesContainer.innerHTML = '';
            
            chat.messages.forEach(msg => {
                const div = document.createElement('div');
                div.className = `flex ${msg.type === 'sent' ? 'justify-end' : 'justify-start'}`;
                div.innerHTML = `
                    <div class="${msg.type === 'sent' ? 'message-sent' : 'message-received'} px-5 py-2 max-w-[70%]">
                        <p>${msg.text}</p>
                        <div class="text-xs text-right mt-1 opacity-70">${msg.time}</div>
                    </div>
                `;
                messagesContainer.appendChild(div);
            });
            
            // Scroll to bottom
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            // Show info panel on large screens
            if (window.innerWidth > 1280) {
                document.getElementById('info-panel').classList.remove('hidden');
                document.getElementById('info-panel').classList.add('flex');
            }
        }

        function sendMessage() {
            if (currentChat === -1) return;
            
            const input = document.getElementById('message-input');
            const text = input.value.trim();
            if (!text) return;
            
            const chat = chats[currentChat];
            const now = new Date();
            const time = `${now.getHours()}:${now.getMinutes().toString().padStart(2, '0')}`;
            
            chat.messages.push({ type: "sent", text: text, time: time });
            
            // Re-render messages
            openChat(currentChat);
            input.value = '';
        }

        function closeChat() {
            currentChat = -1;
            document.getElementById('messages').innerHTML = `
                <div class="flex flex-col items-center justify-center h-full text-[#8696a0]">
                    <div class="text-7xl mb-6">💬</div>
                    <h2 class="text-2xl font-medium">WhatsApp Web</h2>
                    <p class="text-center mt-2 max-w-xs">Send and receive messages without keeping your phone online.<br>Use WhatsApp on up to 4 linked devices.</p>
                </div>
            `;
            document.getElementById('header-name').textContent = 'Select a chat';
            document.getElementById('header-status').textContent = '';
        }

        function newChat() {
            alert("🆕 New chat flow would open here (real WhatsApp behavior)");
        }

        function toggleInfoPanel() {
            const panel = document.getElementById('info-panel');
            panel.classList.toggle('hidden');
            panel.classList.toggle('flex');
        }

        function toggleStatus() {
            alert("👤 Your profile & status settings (real WhatsApp)");
        }

        function toggleMenu() {
            alert("⋯ More options menu (archive, mute, etc.)");
        }

        // Auto dark mode (WhatsApp is always dark by default)
        console.log('%c✅ WhatsApp Web Desktop UI cloned perfectly!', 'color:#25d366; font-size:16px; font-weight:bold');

        // Keyboard support
        document.addEventListener('keydown', (e) => {
            if (e.key === '/' && document.activeElement.tagName !== "INPUT") {
                document.getElementById('search-input').focus();
                e.preventDefault();
            }
        });
    </script>
